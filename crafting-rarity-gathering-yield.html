<script>
/* ── TAB SWITCHING ── */
function switchTab(name, btn) {
document.querySelectorAll('.tab-content').forEach(el => el.classList.remove('active'));
document.querySelectorAll('.tab-btn').forEach(el => el.classList.remove('active'));
document.getElementById('tab-' + name).classList.add('active');
btn.classList.add('active');
}

/* ══════════════════════════════════════════════════
RARITY CALCULATOR
══════════════════════════════════════════════════ */
const RARITY_COLORS = {
Epic:'#a335ee', Rare:'#0070dd', Uncommon:'#1eff00',
Common:'#ffffff', Poor:'#9d9d9d', Failure:'#ff4444'
};

function calcRarities(D) {
let rangeMin, rangeMax;
if (D >= 0) {
rangeMin = Math.min(1 + 4 * D, 180);
rangeMax = 180;
} else {
rangeMin = 1;
rangeMax = Math.max(60, 180 + 4 * D);
}
const rangeSize = rangeMax - rangeMin + 1;
function getChance(minScore, maxScore) {
const overlapMin = Math.max(rangeMin, minScore);
const overlapMax = Math.min(rangeMax, maxScore);
if (overlapMin > overlapMax) return 0;
return ((overlapMax - overlapMin + 1) / rangeSize) * 100;
}
return {
Epic: getChance(166,999), Rare: getChance(146,165),
Uncommon: getChance(121,145), Common: getChance(91,120),
Poor: getChance(51,90), Failure: getChance(-999,50),
rangeMin, rangeMax
};
}

function renderRarity() {
const skill = parseInt(document.getElementById('r-skillLvl').value) || 1;
const tool = parseInt(document.getElementById('r-toolLvl').value) || 1;
const event_b = parseInt(document.getElementById('r-eventBonus').value) || 0;
const atelier = parseInt(document.getElementById('r-atelierLvl').value) || 0;
const D = skill + event_b + atelier - tool;
const r = calcRarities(D);

const dClass = D >= 0 ? 'd-pos' : 'd-neg';
document.getElementById('r-dInfo').innerHTML =
`<b>D</b> = SkillLvl(<span>${skill}</span>) + EventBonus(<span>${event_b}</span>) + AtelierLvl(<span>${atelier}</span>) − ToolLvl(<span>${tool}</span>) = <span class="${dClass}">${D}</span>` +
`<br>Roll range: <span>${r.rangeMin}</span> – <span>${r.rangeMax}</span> (${r.rangeMax - r.rangeMin + 1} values)`;

const barsEl = document.getElementById('r-rarityBars');
barsEl.innerHTML = '';
for (const [name, pct] of Object.entries(r)) {
if (name === 'rangeMin' || name === 'rangeMax') continue;
const row = document.createElement('div');
row.className = 'rarity-row';
row.innerHTML = `
<div class="rarity-label" style="color:${RARITY_COLORS[name]}">${name}</div>
<div class="bar-bg"><div class="bar-fill" style="width:${pct.toFixed(2)}%;background:${RARITY_COLORS[name]};opacity:0.85"></div></div>
<div class="rarity-pct">${pct.toFixed(1)}%</div>`;
barsEl.appendChild(row);
}

const maxLvl = Math.max(tool + 10, 50);
const tbody = document.getElementById('r-tableBody');
tbody.innerHTML = '';
for (let lvl = 1; lvl <= maxLvl; lvl++) {
const d = skill + event_b + atelier - lvl;
const rr = calcRarities(d);
const tr = document.createElement('tr');
if (lvl === tool) tr.className = 'highlight-row';
tr.innerHTML = `
<td class="lvl-col">${lvl}</td>
<td style="color:${d>=0?'#4caf50':'#f44336'}">${d}</td>
<td class="epic-c">${rr.Epic.toFixed(1)}%</td>
<td class="rare-c">${rr.Rare.toFixed(1)}%</td>
<td class="uncom-c">${rr.Uncommon.toFixed(1)}%</td>
<td class="com-c">${rr.Common.toFixed(1)}%</td>
<td class="poor-c">${rr.Poor.toFixed(1)}%</td>
<td class="fail-c">${rr.Failure.toFixed(1)}%</td>`;
tbody.appendChild(tr);
}
}

document.querySelectorAll('#tab-rarity input').forEach(el => el.addEventListener('input', renderRarity));

/* ══════════════════════════════════════════════════
GATHERING YIELD CALCULATOR
══════════════════════════════════════════════════ */
const G_RARITY_BASES = { Poor:-0.15, Common:0.00, Uncommon:0.40, Rare:0.80, Epic:1.20, Legendary:1.50 };
const G_RARITY_MULTS = { Poor:0.95, Common:1.00, Uncommon:1.05, Rare:1.10, Epic:1.15, Legendary:1.20 };
const G_RARITY_COLORS = { Poor:'#9d9d9d', Common:'#ffffff', Uncommon:'#1eff00', Rare:'#0070dd', Epic:'#a335ee', Legendary:'#ff8000' };

function calcPower(toolLvl, rarity, enhance, durability) {
const rBase = G_RARITY_BASES[rarity] ?? 0;
const rMult = G_RARITY_MULTS[rarity] ?? 1;
const base = toolLvl + rBase + (0.25 * enhance);
const mult = rMult + (0.10 * enhance);
const power = base * mult;
return { base, mult, power, effectivePower: power * durability };
}

function calcYield(effectivePower, skillLvl, glyph, masteryPower, synergyBonus, foodBuff, starving, buildingLvl, tithe) {
let y = Math.floor((effectivePower * 2) * skillLvl);
const steps = [];
steps.push({ label: 'Raw Yield = floor(EffPower × 2 × SkillLvl)', val: y });

if (glyph === 'glyph_of_laborer') { y = Math.floor(y * 1.20); steps.push({ label: 'Glyph of Laborer ×1.20', val: y, tag:'glyph', tagLabel:'+20%' }); }
if (glyph === 'glyph_of_prospector') { y = Math.floor(y * 0.95); steps.push({ label: 'Glyph of Prospector ×0.95', val: y, tag:'glyph', tagLabel:'−5%' }); }
if (glyph === 'glyph_of_scavenging') { y = Math.floor(y * 0.90); steps.push({ label: 'Glyph of Scavenging ×0.90', val: y, tag:'glyph', tagLabel:'−10%' }); }
if (glyph === 'glyph_of_philanthropist'){ y = Math.floor(y * 0.95); steps.push({ label: 'Glyph of Philanthropist ×0.95', val: y, tag:'glyph', tagLabel:'−5%' }); }

if (masteryPower > 0) { y = Math.floor(y * (1 + masteryPower * 0.01)); steps.push({ label: `Mastery ×${(1+masteryPower*0.01).toFixed(4)}`, val: y, tag:'buff', tagLabel:`+${masteryPower}%` }); }
if (synergyBonus > 0) { y = Math.floor(y * (1 + synergyBonus/100)); steps.push({ label: `Synergy ×${(1+synergyBonus/100).toFixed(4)}`, val: y, tag:'buff', tagLabel:`+${synergyBonus}%` }); }
if (starving) { y = Math.floor(y * 0.75); steps.push({ label: 'Starving ×0.75', val: y, tag:'debuff', tagLabel:'−25%' }); }
if (foodBuff) { y = Math.floor(y * 1.25); steps.push({ label: 'Food Buff ×1.25', val: y, tag:'buff', tagLabel:'+25%' }); }
if (buildingLvl > 0) { y = Math.floor(y * (1 + buildingLvl*0.01)); steps.push({ label: `Community Bldg Lvl ${buildingLvl} ×${(1+buildingLvl*0.01).toFixed(2)}`, val: y, tag:'buff', tagLabel:`+${buildingLvl}%` }); }
if (tithe > 0) { const amt = y*(tithe/100); y = Math.floor(y-amt); steps.push({ label: `Tithe ${tithe}%`, val: y, tag:'debuff', tagLabel:`−${tithe}%` }); }

return { final: y, steps };
}

function renderGather() {
const toolType = document.getElementById('g-toolType').value;
const toolLvl = parseInt(document.getElementById('g-toolLvl').value) || 1;
const rarity = document.getElementById('g-rarity').value;
const enhance = parseInt(document.getElementById('g-enhance').value) || 0;
const durability = parseFloat(document.getElementById('g-durability').value) ?? 1;
const skillLvl = parseInt(document.getElementById('g-skillLvl').value) || 1;
const glyph = document.getElementById('g-glyph').value;
const masteryPow = parseFloat(document.getElementById('g-masteryPower').value) || 0;
const synergy = parseFloat(document.getElementById('g-synergyBonus').value) || 0;
const buildingLvl = parseInt(document.getElementById('g-buildingLvl').value) || 0;
const tithe = parseFloat(document.getElementById('g-tithe').value) || 0;
const foodBuff = document.getElementById('g-foodBuff').checked;
const starving = document.getElementById('g-starving').checked;

const pw = calcPower(toolLvl, rarity, enhance, durability);
const res = calcYield(pw.effectivePower, skillLvl, glyph, masteryPow, synergy, foodBuff, starving, buildingLvl, tithe);
const rarityColor = G_RARITY_COLORS[rarity];

document.getElementById('g-formulaBox').innerHTML = `
<b>Step 1 — Tool Power</b><br>
base = ToolLvl(<span>${toolLvl}</span>) + RarityBase(<span style="color:${rarityColor}">${G_RARITY_BASES[rarity]}</span>) + 0.25×Enhance(<span>${enhance}</span>) = <span>${pw.base.toFixed(3)}</span><br>
mult = RarityMult(<span style="color:${rarityColor}">${G_RARITY_MULTS[rarity]}</span>) + 0.10×Enhance(<span>${enhance}</span>) = <span>${pw.mult.toFixed(3)}</span><br>
Power = base × mult = <span>${pw.power.toFixed(4)}</span> &nbsp;|&nbsp; EffectivePower = Power × Durability(<span>${durability}</span>) = <span>${pw.effectivePower.toFixed(4)}</span><br><br>
<b>Step 2 — Raw Yield</b><br>
floor(EffPower × 2 × SkillLvl) = floor(<span>${pw.effectivePower.toFixed(4)}</span> × 2 × <span>${skillLvl}</span>) = <span>${Math.floor(pw.effectivePower*2*skillLvl).toLocaleString()}</span><br><br>
<b>Step 3 — Modifiers</b> (each applied with floor)
`;

const bEl = document.getElementById('g-breakdown');
bEl.innerHTML = '';
for (const s of res.steps) {
const tagHtml = s.tag ? `<span class="tag tag-${s.tag}">${s.tagLabel}</span>` : '';
const isLast = s === res.steps[res.steps.length - 1];
const div = document.createElement('div');
div.className = 'brow' + (isLast ? ' highlight' : '');
div.innerHTML = `<span class="blabel">${s.label}${tagHtml}</span><span class="bval">${s.val.toLocaleString()}</span>`;
bEl.appendChild(div);
}

document.getElementById('g-resultBig').innerHTML = `${res.final.toLocaleString()} <small>per gather tick</small>`;

const tbody = document.getElementById('g-tableBody');
tbody.innerHTML = '';
const maxLvl = Math.max(toolLvl + 10, 30);
for (let lvl = 1; lvl <= maxLvl; lvl++) {
const p = calcPower(lvl, rarity, enhance, durability);
const r = calcYield(p.effectivePower, skillLvl, glyph, masteryPow, synergy, foodBuff, starving, buildingLvl, tithe);
const rawYield = Math.floor(p.effectivePower * 2 * skillLvl);
const tr = document.createElement('tr');
if (lvl === toolLvl) tr.className = 'highlight-row';
tr.innerHTML = `
<td class="lvl-col">${lvl}</td>
<td>${p.power.toFixed(3)}</td>
<td>${p.effectivePower.toFixed(3)}</td>
<td>${rawYield.toLocaleString()}</td>
<td><b>${r.final.toLocaleString()}</b></td>`;
tbody.appendChild(tr);
}

}

document.querySelectorAll('#tab-gather input, #tab-gather select').forEach(el => el.addEventListener('input', renderGather));

/* ── INIT ── */
renderRarity();
renderGather();
